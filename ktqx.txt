         function playRollSound(duration) {
             initAudio();
             if(!audioCtx) return;
         
             let bufferSize = audioCtx.sampleRate * duration;
             let buffer = audioCtx.createBuffer(1, bufferSize, audioCtx.sampleRate);
             let data = buffer.getChannelData(0);
             for (let i = 0; i < bufferSize; i++) {
                 data[i] = Math.random() * 2 - 1;
             }
         
             let noiseNode = audioCtx.createBufferSource();
             noiseNode.buffer = buffer;
             
             let filter = audioCtx.createBiquadFilter();
             filter.type = 'bandpass';
             filter.frequency.setValueAtTime(480, audioCtx.currentTime);
             filter.Q.setValueAtTime(4.0, audioCtx.currentTime);
         
             let lfo = audioCtx.createOscillator();
             lfo.type = 'sine';
             lfo.frequency.setValueAtTime(16, audioCtx.currentTime); 
             let lfoGain = audioCtx.createGain();
             lfoGain.gain.setValueAtTime(200, audioCtx.currentTime);
         
             lfo.connect(lfoGain);
             lfoGain.connect(filter.frequency);
         
             let gainNode = audioCtx.createGain();
             gainNode.gain.setValueAtTime(0.45, audioCtx.currentTime); 
             gainNode.gain.exponentialRampToValueAtTime(0.001, audioCtx.currentTime + duration);
         
             noiseNode.connect(filter);
             filter.connect(gainNode);
             gainNode.connect(audioCtx.destination);
             
             lfo.start();
             noiseNode.start();
             lfo.stop(audioCtx.currentTime + duration);
             noiseNode.stop(audioCtx.currentTime + duration);
         }

         initAudio();
         if (!audioCtx) return;
         
         const now = audioCtx.currentTime;
         
         // 模拟骰子连续碰撞
         const hits = 2 + Math.floor(Math.random() * 3);
         
         for (let i = 0; i < hits; i++) {
         
             const osc = audioCtx.createOscillator();
             const gain = audioCtx.createGain();
         
             const freq =
                 (120 + Math.random() * 180) *
                 pitchFactor;
         
             // 方波更像塑料骰子碰撞
             osc.type = 'square';
         
             osc.frequency.setValueAtTime(
                 freq,
                 now + i * 0.03
             );
         
             osc.frequency.exponentialRampToValueAtTime(
                 freq * 2.2,
                 now + i * 0.03 + 0.015
             );
         
             gain.gain.setValueAtTime(
                 0.18,
                 now + i * 0.03
             );
         
             gain.gain.exponentialRampToValueAtTime(
                 0.001,
                 now + i * 0.03 + 0.04
             );
         
             osc.connect(gain);
             gain.connect(audioCtx.destination);
         
             osc.start(now + i * 0.03);
             osc.stop(now + i * 0.05);
         }

         const stopOsc = audioCtx.createOscillator();
         const stopGain = audioCtx.createGain();

             }
             if (hintTxt) hintTxt.remove();
             
             var parentRowIdx = Math.floor(totalRolled / currentLineCapacity);
             var targetRowDom = document.getElementById('dice-row-' + parentRowIdx);
             
             let dynamicMaxDuration = 0;
             let finishedCount = 0;
         
             playRollSound(3.0);
         
             // 【核心重构：第一步】开始转动前，先在对应队列中抢占坑位，防止顺序错乱
             var currentRollStartIndex = totalRolled;
             for (let i = 0; i < numToRoll; i++) {
                 activeDiceRenderQueue[currentRollStartIndex + i] = { status: "rolling" }; 
             }
             renderPipsFlow(); // 先把占位符刷出来
         
             for (let i = 0; i < numToRoll; i++) {
                 // 精确计算这颗骰子对应的全局索引位置
                 let absoluteDiceIndex = currentRollStartIndex + i; 
         
                 let targetPoint = Math.floor(Math.random() * 6) + 1;
                 
                 let duration = 1.5 + Math.random() * 1.0; 
                 let delay = Math.random() * 0.15;          
                 let totalTime = (duration + delay) * 1000;
                 if(totalTime > dynamicMaxDuration) {
                     dynamicMaxDuration = totalTime;
                 }
         
                 let containerDom = document.createElement('div');
                 containerDom.className = 'cube-container';
                 
                 let shadowDom = document.createElement('div');
                 shadowDom.className = 'cube-shadow';
                 containerDom.appendChild(shadowDom);

                     let faceDom = document.createElement('div');
                     faceDom.className = 'cube-face face-' + faceNum;
                     let pipsArr = facePipsConfig[faceNum];

                         let pipDom = document.createElement('div');
                         pipDom.className = 'pip ' + pipsArr[pIdx];
                         faceDom.appendChild(pipDom);
                     }
                     cubeDom.appendChild(faceDom);
                 }
                 
                 containerDom.appendChild(cubeDom);
                 if(targetRowDom) targetRowDom.appendChild(containerDom);
         
                 let baseAngle = faceTransforms[targetPoint];
                 let rotX = baseAngle.x + (Math.floor(Math.random() * 3) + 3) * 360;
                 let rotY = baseAngle.y + (Math.floor(Math.random() * 3) + 3) * 360;
                 let rotZ = baseAngle.z + Math.floor(Math.random() * 2) * 180;
         
                 setTimeout(() => {
                     cubeDom.style.transform = `rotateX(${rotX}deg) rotateY(${rotY}deg) rotateZ(${rotZ}deg)`;
                     shadowDom.style.transform = 'scale(1.5)';
                     shadowDom.style.background = 'rgba(0,0,0,0.15)';
                     shadowDom.style.filter = 'blur(5px)';
                 }, 20);
         
                 // 落地稳固落定
                 setTimeout(() => {
                     cubeDom.classList.add('impact-effect');
                     shadowDom.style.transform = 'scale(0.95)';
                     shadowDom.style.background = 'rgba(0,0,0,0.85)';
                     shadowDom.style.filter = 'blur(2px)';
                     
                     playClickSound(0.9 + (targetPoint * 0.1));
         
                     let primaryFace = cubeDom.querySelector('.face-1');
                     if (primaryFace && targetPoint !== 1) {
                         primaryFace.innerHTML = ''; 
                         let newPips = facePipsConfig[targetPoint];
                         newPips.forEach(pClass => {
                             let pDiv = document.createElement('div');
                             pDiv.className = 'pip ' + pClass;
                             pDiv.style.backgroundColor = '#111';
                             pDiv.style.width = '9px';
                             pDiv.style.height = '9px';
                             primaryFace.appendChild(pDiv);
                         });
                     }
         
                     // 【核心重构：第二步】定格瞬间，依据强绑定的物理索引精准回填，不打乱整体队伍顺序！
                     activeDiceRenderQueue[absoluteDiceIndex] = targetPoint;
                     calculatePrices();
         
                     finishedCount++;
                     if (finishedCount === numToRoll) {
                         if(totalRolled >= maxTargetDice) {
                             document.getElementById('try-status-bar').innerHTML = "✨骰子数：室内挂机5颗，方形柜机10颗，天花机7颗";
                             rollBtn.innerText = "🔄 继续摇一摇";
                             rollBtn.classList.add("reset-state");
                             currentButtonMode = "reset";
                             rollBtn.disabled = false;
                         } else {
                             rollBtn.disabled = false;
                             rollNumSelect.disabled = false;
                         }
                     }
                 }, totalTime + 50); 
             }
             
             totalRolled += numToRoll;
             currentCountSpan.innerText = totalRolled;
         }
